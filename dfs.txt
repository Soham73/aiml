graph = {
'1' : ['2','3'],
'2' : ['4','5'],
'3' : ['6'],
'4' : [],
'5' : [], 
'6' : []
}
visited = set()
def dfs(visited,graph,node):
        if node not in visited:
              print(node,'-',end="")
              visited.add(node)
              for neighbour in graph[node]:
                dfs(visited,graph,neighbour)
print("Following is the DFS Traversal: ")
dfs(visited,graph,'1')

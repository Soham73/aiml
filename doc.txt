import random

# Create a deck of cards
suits = ["Hearts", "Diamonds", "Clubs", "Spades"]
ranks = ["2", "3", "4", "5", "6", "7", "8", "9", "10", "Jack", "Queen",
         "King", "Ace"]
deck = []  # empty list
for suit in suits:
    for rank in ranks:
        card = rank + " of " + suit
        deck.append(card)

print(deck)

print("Original Deck:")
print(deck)

# Shuffle the deck
random.shuffle(deck)

print("\nShuffled Deck:")
print(deck)

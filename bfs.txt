graph = {
'A' : ['B','C'],
'B' : ['D','E'],
'C' : ['F','G'],
'D' : [],
'E' : [],
'F' : [],
'G' : []
}
print(graph)
queue = []
def mybfs(graph,queue,initialnode):
        queue.append(initialnode)
        while queue:
            m = queue.pop(0)
            print(m,'=',end="")
            for neighbour in graph[m]:
                queue.append(neighbour)
print("Traversal with BFS")
mybfs(graph,queue,'A')
